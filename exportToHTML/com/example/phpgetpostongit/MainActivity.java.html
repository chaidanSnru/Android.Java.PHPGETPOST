10   import java.io.IOException;

12   import okhttp3.Call;
13   import okhttp3.Callback;
14   import okhttp3.OkHttpClient;
15   import okhttp3.Request;
16   import okhttp3.Response;
17   
18   public class MainActivity extends AppCompatActivity {
19   
20       @Override
21       protected void onCreate(Bundle savedInstanceState) {
22           super.onCreate(savedInstanceState);
23           setContentView(R.layout.activity_main);
24       }
25   
26       public void onClick(View view) {
27           final TextView txtResult = (TextView) findViewById(R.id.textView);
28           // Instantiate new OkHttpClient
29           OkHttpClient client = new OkHttpClient();
30   
31           // Initialize a new Request
32           Request request = new Request.Builder()
33                   //.url("https://chaidan2.000webhostapp.com/first.php")
34                   .url("https://chaidan2.000webhostapp.com/userget.php/?username=admin&password=admin")
35                   .build();
36   
37           client.newCall(request).enqueue(new Callback() {
38               @Override
39               public void onFailure(Call call, IOException e) {
40                   // Do something when request failed
41                   e.printStackTrace();
42                   Log.d("OKhttp", "Request Failed.");
43               }
44   
45               @Override
46               public void onResponse(Call call, Response response) throws IOException {
47                   if (!response.isSuccessful()) {
48                       throw new IOException("Error : " + response);
49                   } else {
50                       Log.d("OKhttp", "Request Successful.");
51                   }
52   
53                   // Read data in the worker thread
54                   final String data = response.body().string();
55   
56                   // Display the requested data on UI in main thread
57                   runOnUiThread(new Runnable() {
58                       @Override
59                       public void run() {
60                           // Display requested url data as string into text view
61                           txtResult.setText(data);
62                       }
63                   });
64               }
65           });
66       }
67   }
